**4. 换乘站识别**
```javascript
function getTransferStations() {
  const stationLines = {};
  metroLines.forEach(line => {
    line.stations.forEach(station => {
      if (!stationLines[station]) stationLines[station] = [];
      stationLines[station].push(line);
    });
  });
  return Object.entries(stationLines)
    .filter(([station, lines]) => lines.length > 1)
    .map(([station, lines]) => ({
      name: station,
      lines: lines,
      lineCount: lines.length
    }));
}
```
